#!/usr/bin/python3
import unittest
from say_my_name import say_my_name

class TestSayMyName(unittest.TestCase):

    def test_say_my_name_full_name(self):
        self.assertEqual(say_my_name("John", "Smith"), "My name is John Smith")

    def test_say_my_name_first_name_only(self):
        self.assertEqual(say_my_name("Bob"), "My name is Bob ")

    def test_say_my_name_empty_last_name(self):
        self.assertEqual(say_my_name("Alice", ""), "My name is Alice ")

    def test_say_my_name_non_string_first_name(self):
        with self.assertRaises(TypeError):
            say_my_name(12, "Smith")

    def test_say_my_name_non_string_last_name(self):
        with self.assertRaises(TypeError):
            say_my_name("John", 12)

    def test_say_my_name_no_args(self):
        with self.assertRaises(TypeError):
            say_my_name()

if __name__ == '__main__':
    unittest.main()
